#shadervertex
precision mediump float;

attribute vec2 vertPosition;
attribute vec3 vertColor;
varying vec3 fragColor;

void main()
{
    fragColor = vertColor;
    gl_Position = vec4(vertPosition, 0.0, 1.0);
}

#shaderfragment
precision mediump float;

varying vec3 fragColor;

void main()
{
    gl_FragColor = vec4(fragColor,1.0);
}
